DATA SEGMENT

    COUNT EQU 0007H
    NEXTFIB DW 0001H 
    LIST DW 0000H, 0001H

DATA ENDS

DISPLAY MACRO CHAR 
  MOV AH, 02H 
  LEA DX, CHAR
  INT 21H
ENDM

CODE SEGMENT

  ASSUME CS:CODE,DS:DATA
  START:
          
  MOV AX, DATA
  MOV DS, AX
  MOV CX, COUNT
  MOV AX, 0000H 
  LEA SI, LIST   

L1:    
  MOV BX, AX
  ADD AX, NEXTFIB
  MOV NEXTFIB, BX
  MOV [SI+2], AX   
  INC SI
  INC SI
  LOOP L1

EXIT:
  MOV AH, 4CH
  INT 21H
           
CODE ENDS    
END START
